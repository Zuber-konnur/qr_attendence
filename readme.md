-- 2. Create or Update Users Table
CREATE TABLE IF NOT EXISTS users (
    usn TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    domain TEXT,
    batch TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 3. Create or Update Attendance Table
CREATE TABLE IF NOT EXISTS attendance (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    usn TEXT REFERENCES users(usn),
    date DATE NOT NULL DEFAULT CURRENT_DATE,
    login_time TIME,
    lunch_start TIME,
    lunch_end TIME,
    logout_time TIME,
    status TEXT DEFAULT 'On Duty',
    remarks TEXT,
    device_id TEXT,
    UNIQUE(usn, date)
);

-- Safely add Phase 2 columns if the table already existed from Phase 1

ADD COLUMN IF NOT EXISTS lunch_start TIME,
ADD COLUMN IF NOT EXISTS lunch_end TIME,
ADD COLUMN IF NOT EXISTS remarks TEXT;
